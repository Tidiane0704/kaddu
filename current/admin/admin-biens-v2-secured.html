Chargement…
⚠️ Exécutez ces blocs SQL une seule fois dans Supabase → SQL Editor → New Query. Respectez l'ordre.
1. Table commerciaux (avant biens car FK)
CREATE TABLE IF NOT EXISTS commerciaux ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, nom TEXT NOT NULL, role TEXT, telephone TEXT, whatsapp TEXT, email TEXT, description TEXT, statut TEXT NOT NULL DEFAULT 'actif', photo_url TEXT, photo_path TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() );
2. Table biens
CREATE TABLE IF NOT EXISTS biens ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, titre TEXT NOT NULL, slug TEXT UNIQUE NOT NULL, ref_interne TEXT, type_bien TEXT DEFAULT 'appartement', categorie TEXT NOT NULL DEFAULT 'meuble', statut TEXT NOT NULL DEFAULT 'brouillon', accroche TEXT, adresse TEXT, quartier TEXT, ville TEXT DEFAULT 'Dakar', immeuble TEXT, lat NUMERIC, lng NUMERIC, zone_carte TEXT, desc_quartier TEXT, prix TEXT, prix_sub TEXT, charges TEXT, caution TEXT, frais_dossier TEXT, disponibilite DATE, duree_min TEXT, type_bail TEXT, paiements TEXT, checkin TEXT, checkout TEXT, commercial_id UUID REFERENCES commerciaux(id) ON DELETE SET NULL, surface INTEGER, pieces TEXT, chambres INTEGER, sdb INTEGER, etage TEXT, capacite INTEGER, nb_etages INTEGER, annee INTEGER, equipements_bool JSONB DEFAULT '{}'::jsonb, equip_libres TEXT, description TEXT, atouts TEXT, conditions TEXT, visite_360 TEXT, video TEXT, pdf_url TEXT, photos JSONB DEFAULT '[]'::jsonb, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() );
3. RLS Policies
ALTER TABLE biens ENABLE ROW LEVEL SECURITY; ALTER TABLE commerciaux ENABLE ROW LEVEL SECURITY; -- Lecture publique des biens disponibles CREATE POLICY "biens_public_read" ON biens FOR SELECT USING (statut != 'brouillon'); CREATE POLICY "biens_admin_all" ON biens FOR ALL USING (auth.role() = 'authenticated'); -- Lecture publique des commerciaux actifs CREATE POLICY "comm_public_read" ON commerciaux FOR SELECT USING (statut = 'actif'); CREATE POLICY "comm_admin_all" ON commerciaux FOR ALL USING (auth.role() = 'authenticated');
4. Storage bucket biens-photos (déjà créé ✅)
-- Policies sur le bucket biens-photos existant : INSERT INTO storage.policies (name, bucket_id, operation, definition) VALUES ('biens_public_read', 'biens-photos', 'SELECT', 'true'), ('biens_admin_write', 'biens-photos', 'INSERT', 'auth.role() = ''authenticated'''), ('biens_admin_del', 'biens-photos', 'DELETE', 'auth.role() = ''authenticated''') ON CONFLICT DO NOTHING; -- Alternative si la syntaxe ci-dessus ne fonctionne pas, via l'UI Storage > Policies

✦ Le bucket biens-photos est déjà créé (visible dans votre capture). ✦ Une fois le SQL exécuté, revenez sur l'onglet Biens.
